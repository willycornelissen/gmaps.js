      var items, markers_data = [];

          var item = items[i];

          if (item.location.lat != undefined && item.location.lng != undefined) {

              lat : item.location.lat,
              lng : item.location.lng,
              title : item.name,
              icon : {
                size : new google.maps.Size(32, 32),
                url : icon
              }
            });
          }
        }
      }

      map.addMarkers(markers_data);
    }
